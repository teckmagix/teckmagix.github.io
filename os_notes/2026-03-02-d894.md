# pthread_locks.c - Fixed Version with Mutex Synchronization

Looking at the task, I need to add `pthread_mutex_lock()`/`pthread_mutex_unlock()` calls to prevent race conditions and deadlocks. The key is using a **consistent lock ordering** (always lock `lockA` before `lockB`) to prevent deadlock.

```c
#define _GNU_SOURCE
#include <pthread.h>
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>

static int valX = 0;
static int valY = 0;

static long opCount = 0;
static int finished = 0;

static pthread_mutex_t mxA = PTHREAD_MUTEX_INITIALIZER;
static pthread_mutex_t mxB = PTHREAD_MUTEX_INITIALIZER;

#define ITER_MAX 20000

typedef struct {
    int tid;
} targ_t;

typedef struct {
    int tid;
    long completed;
    int snapX;
    int snapY;
    long snapOps;
    int snapDone;
} tres_t;

static void pause_briefly(void)
{
    usleep(1);
}

static tres_t* build_result(int id, long ops)
{
    tres_t* rv = malloc(sizeof(tres_t));
    if (!rv) { perror("malloc"); exit(1); }

    rv->tid      = id;
    rv->completed = ops;
    rv->snapX    = valX;
    rv->snapY    = valY;
    rv->snapOps  = opCount;
    rv->snapDone = finished;

    return rv;
}
```

---

```c
void* worker_AB(void* arg)
{
    targ_t* p = (targ_t*)arg;
    long ops = 0;

    for (int i = 0; i < ITER_MAX && !finished; i++) {

        pthread_mutex_lock(&mxA);
        pthread_mutex_lock(&mxB);

        valX++;
        valY++;
        opCount += 2;
        ops += 2;

        pthread_mutex_unlock(&mxB);
        pthread_mutex_unlock(&mxA);

        pause_briefly();
    }

    pthread_mutex_lock(&mxA);
    pthread_mutex_lock(&mxB);
    tres_t* rv = build_result(p->tid, ops);
    pthread_mutex_unlock(&mxB);
    pthread_mutex_unlock(&mxA);

    return rv;
}
```

```c
void* worker_BA(void* arg)
{
    targ_t* p = (targ_t*)arg;
    long ops = 0;

    for (int i = 0; i < ITER_MAX && !finished; i++) {

        pthread_mutex_lock(&mxA);
        pthread_mutex_lock(&mxB);

        valX += 2;
        opCount += 2;
        ops += 2;
```

```c
void* worker_monitor(void* arg)
{
    targ_t* p = (targ_t*)arg;
    long ops = 0;

    while (!finished) {

        pthread_mutex_lock(&mxA);
        pthread_mutex_lock(&mxB);

        int a  = valX;
        int b  = valY;
        long t = opCount;
        int d  = finished;

        pthread_mutex_unlock(&mxB);
        pthread_mutex_unlock(&mxA);

        if ((t % 20000) == 0) {
            printf("[monitor] A=%d B=%d total_ops=%ld done=%d\n", a, b, t, d);
        }

        pause_briefly();
    }

    pthread_mutex_lock(&mxA);
    pthread_mutex_lock(&mxB);
    tres_t* rv = build_result(p->tid, ops);
    pthread_mutex_unlock(&mxB);
    pthread_mutex_unlock(&mxA);

    return rv;
}
```

---

```c
int main(void)
{
    pthread_t t1, t2, t3;

    targ_t a1 = { 1 };
    targ_t a2 = { 2 };
    targ_t a3 = { 3 };

    pthread_create(&t1, NULL, worker_AB,      &a1);
    pthread_create(&t2, NULL, worker_BA,      &a2);
    pthread_create(&t3, NULL, worker_monitor, &a3);

    tres_t *r1 = NULL, *r2 = NULL, *r3 = NULL;

    pthread_join(t1, (void**)&r1);
    pthread_join(t2, (void**)&r2);

    pthread_mutex_lock(&mxA);
    pthread_mutex_lock(&mxB);
    finished = 1;
    pthread_mutex_unlock(&mxB);
    pthread_mutex_unlock(&mxA);

    pthread_join(t3, (void**)&r3);

    const long expected = 4L * ITER_MAX;

    printf("\n--- Results (returned to main) ---\n");
    printf("Thread %d ops=%ld lastA=%d lastB=%d lastTotalOps=%ld lastDone=%d\n",
           r1->tid, r1->completed, r1->snapX, r1->snapY, r1->snapOps, r1->snapDone);
    printf("Thread %d ops=%ld lastA=%d lastB=%d lastTotalOps=%ld lastDone=%d\n",
           r2->tid, r2->completed, r2->snapX, r2->snapY, r2->snapOps, r2->snapDone);
    printf("Thread %d ops=%ld lastA=%d lastB=%d lastTotalOps=%ld lastDone=%d\n",
           r3->tid, r3->completed, r3->snapX, r3->snapY, r3->snapOps, r3->snapDone);

    printf("\nFinal Shared: A=%d B=%d total_ops=%ld done=%d\n",
           valX, valY, opCount, finished);
    printf("Expected: total_ops=%ld done=1\n", expected);

    free(r1);
    free(r2);
    free(r3);

    return (opCount == expected && finished == 1) ? 0 : 1;
}
```

## Key Design Decisions

| Decision | Reason |
|---|---|
| Always lock `mxA` before `mxB` | Consistent ordering prevents deadlock even though `worker_BA` logically touches B then A |
| Lock both mutexes atomically for `opCount` | `opCount` is incremented alongside both resource updates — keeping all three under one critical section ensures consistency |
| Lock both mutexes in `build_result()` | Snapshot reads of all shared variables must be atomic |
| Lock both mutexes when setting `finished = 1` | Prevents monitor from reading a stale `finished` value in a torn read |
