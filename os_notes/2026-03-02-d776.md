// pthread_locks.c
//
// Compile: gcc -O2 -Wall -Wextra -pthread pthread_locks.c -o student
// Run: ./ pthread_locks
//
// - Deterministic end state
//     done == 1
//     total_ops == 4 * LOOP_COUNT
//
// To identify and ADD pthread_mutex_lock()/pthread_mutex_unlock() calls ONLY.
// Do not change the algorithm.

#define _GNU_SOURCE
#include <pthread.h>
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>

// -------------------- Shared Global Resources --------------------

static int resX = 0;
static int resY = 0;

static long total_count = 0;
static int finished = 0;

// Provided mutexes (must use ONLY these)
static pthread_mutex_t lockA = PTHREAD_MUTEX_INITIALIZER;
static pthread_mutex_t lockB = PTHREAD_MUTEX_INITIALIZER;

#define LOOP_COUNT 20000

// -------------------- Thread Argument / Result Types --------------------

typedef struct {
    int thread_id;
} targs_t;

typedef struct {
    int thread_id;
    long ops_done;
    int lastX;
    int lastY;
    long lastTotalCount;
    int lastFinished;
} tresult_t;

static void pause_briefly(void)
{
    usleep(1);
}

static tresult_t* build_result(int id, long ops)
{
    tresult_t* r = malloc(sizeof(tresult_t));
    if (!r) { perror("malloc"); exit(1); }

    r->thread_id = id;
    r->ops_done = ops;

    pthread_mutex_lock(&lockA);
    pthread_mutex_lock(&lockB);
    r->lastX = resX;
    r->lastY = resY;
    r->lastTotalCount = total_count;
    r->lastFinished = finished;
    pthread_mutex_unlock(&lockB);
    pthread_mutex_unlock(&lockA);

    return r;
}


// -------------------- Worker 1: touches A then B --------------------

void* worker_AB(void* arg)
{
    targs_t* p = (targs_t*)arg;
    long ops = 0;

    for (int i = 0; i < LOOP_COUNT && !finished; i++) {

        pthread_mutex_lock(&lockA);
        pthread_mutex_lock(&lockB);

        resX++;
        resY++;
        total_count += 2;
        ops += 2;

        pthread_mutex_unlock(&lockB);
        pthread_mutex_unlock(&lockA);

        pause_briefly();
    }

    return build_result(p->thread_id, ops);
}


// -------------------- Worker 2: touches B then A (logic order) --------------------

void* worker_BA(void* arg)
{
    targs_t* p = (targs_t*)arg;
    long ops = 0;

    for (int i = 0; i < LOOP_COUNT && !finished; i++) {

        pthread_mutex_lock(&lockA);
        pthread_mutex_lock(&lockB);

        resY += 2;
        resX += 2;
        total_count += 2;
        ops += 2;

        pthread_mutex_unlock(&lockB);
        pthread_mutex_unlock(&lockA);

        pause_briefly();
    }

    return build_result(p->thread_id, ops);
}


// -------------------- Worker 3: monitor (READ-ONLY) --------------------

void* worker_monitor(void* arg)
{
    targs_t* p = (targs_t*)arg;
    long ops = 0;

    while (!finished) {
        pthread_mutex_lock(&lockA);
        pthread_mutex_lock(&lockB);

        int a = resX;
        int b = resY;
        long t = total_count;
        int d = finished;

        pthread_mutex_unlock(&lockB);
        pthread_mutex_unlock(&lockA);

        if ((t % 20000) == 0) {
            printf("[monitor] A=%d B=%d total_ops=%ld done=%d\n", a, b, t, d);
        }

        pause_briefly();
    }

    return build_result(p->thread_id, ops);
}

// -------------------- Main --------------------

int main(void)
{
    pthread_t t1, t2, t3;

    targs_t a1 = { 1 };
    targs_t a2 = { 2 };
    targs_t a3 = { 3 };

    pthread_create(&t1, NULL, worker_AB, &a1);
    pthread_create(&t2, NULL, worker_BA, &a2);
    pthread_create(&t3, NULL, worker_monitor, &a3);

    tresult_t *r1 = NULL, *r2 = NULL, *r3 = NULL;

    // Join workers first
    pthread_join(t1, (void**)&r1);
    pthread_join(t2, (void**)&r2);

    pthread_mutex_lock(&lockA);
    pthread_mutex_lock(&lockB);
    finished = 1;
    pthread_mutex_unlock(&lockB);
    pthread_mutex_unlock(&lockA);

    // Now join monitor
    pthread_join(t3, (void**)&r3);

    const long expected_total_count = 4L * LOOP_COUNT;

    printf("\n--- Results (returned to main) ---\n");
    printf("Thread %d ops=%ld lastA=%d lastB=%d lastTotalOps=%ld lastDone=%d\n",
           r1->thread_id, r1->ops_done, r1->lastX, r1->lastY, r1->lastTotalCount, r1->lastFinished);
    printf("Thread %d ops=%ld lastA=%d lastB=%d lastTotalOps=%ld lastDone=%d\n",
           r2->thread_id, r2->ops_done, r2->lastX, r2->lastY, r2->lastTotalCount, r2->lastFinished);
    printf("Thread %d ops=%ld lastA=%d lastB=%d lastTotalOps=%ld lastDone=%d\n",
           r3->thread_id, r3->ops_done, r3->lastX, r3->lastY, r3->lastTotalCount, r3->lastFinished);

    printf("\nFinal Shared: A=%d B=%d total_ops=%ld done=%d\n",
           resX, resY, total_count, finished);
    printf("Expected: total_ops=%ld done=1\n", expected_total_count);

    free(r1);
    free(r2);
    free(r3);

    return (total_count == expected_total_count && finished == 1) ? 0 : 1;
}
