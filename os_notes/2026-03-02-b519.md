# Solution: `notxv6/pthread_locks.c`

The task requires adding `pthread_mutex_lock()` / `pthread_mutex_unlock()` calls to fix race conditions while avoiding deadlock. The key discipline is: **always acquire lockA before lockB** (consistent lock ordering).

---

```c
#define _GNU_SOURCE
#include <pthread.h>
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>

static int resX = 0;
static int resY = 0;

static long total_count = 0;
static int finished = 0;

static pthread_mutex_t mxA = PTHREAD_MUTEX_INITIALIZER;
static pthread_mutex_t mxB = PTHREAD_MUTEX_INITIALIZER;

#define ITERATIONS 20000

typedef struct {
    int tid;
} targs_t;

typedef struct {
    int tid;
    long ops;
    int snapX;
    int snapY;
    long snapTotal;
    int snapDone;
} tresult_t;

static void delay(void)
{
    usleep(1);
}

static tresult_t* build_result(int id, long ops)
{
    tresult_t* r = malloc(sizeof(tresult_t));
    if (!r) { perror("malloc"); exit(1); }

    r->tid      = id;
    r->ops      = ops;
    r->snapX    = resX;
    r->snapY    = resY;
    r->snapTotal = total_count;
    r->snapDone = finished;

    return r;
}
```

---

```c
void* workerXY(void* arg)
{
    targs_t* p = (targs_t*)arg;
    long ops = 0;

    for (int i = 0; i < ITERATIONS && !finished; i++) {
        pthread_mutex_lock(&mxA);
        pthread_mutex_lock(&mxB);

        resX++;
        resY++;
        total_count += 2;
        ops += 2;

        pthread_mutex_unlock(&mxB);
        pthread_mutex_unlock(&mxA);

        delay();
    }

    pthread_mutex_lock(&mxA);
    pthread_mutex_lock(&mxB);
    tresult_t* r = build_result(p->tid, ops);
    pthread_mutex_unlock(&mxB);
    pthread_mutex_unlock(&mxA);

    return r;
}
```

```c
void* workerMon(void* arg)
{
    targs_t* p = (targs_t*)arg;
    long ops = 0;

    while (!finished) {
        pthread_mutex_lock(&mxA);
        pthread_mutex_lock(&mxB);

        int a  = resX;
        int b  = resY;
        long t = total_count;
        int d  = finished;

        pthread_mutex_unlock(&mxB);
        pthread_mutex_unlock(&mxA);

        if ((t % 20000) == 0) {
            printf("[monitor] A=%d B=%d total_ops=%ld done=%d\n", a, b, t, d);
        }

        delay();
    }

    pthread_mutex_lock(&mxA);
    pthread_mutex_lock(&mxB);
    tresult_t* r = build_result(p->tid, ops);
    pthread_mutex_unlock(&mxB);
    pthread_mutex_unlock(&mxA);

    return r;
}
```

---

```c
int main(void)
{
    pthread_t t1, t2, t3;

    targs_t a1 = { 1 };
    targs_t a2 = { 2 };
    targs_t a3 = { 3 };

    pthread_create(&t1, NULL, workerXY,  &a1);
    pthread_create(&t2, NULL, workerYX,  &a2);
    pthread_create(&t3, NULL, workerMon, &a3);

    tresult_t *r1 = NULL, *r2 = NULL, *r3 = NULL;

    pthread_join(t1, (void**)&r1);
    pthread_join(t2, (void**)&r2);

    pthread_mutex_lock(&mxA);
    pthread_mutex_lock(&mxB);
    finished = 1;
    pthread_mutex_unlock(&mxB);
    pthread_mutex_unlock(&mxA);

    pthread_join(t3, (void**)&r3);

    const long expected = 4L * ITERATIONS;

    printf("\n--- Results (returned to main) ---\n");
    printf("Thread %d ops=%ld lastA=%d lastB=%d lastTotalOps=%ld lastDone=%d\n",
           r1->tid, r1->ops, r1->snapX, r1->snapY, r1->snapTotal, r1->snapDone);
    printf("Thread %d ops=%ld lastA=%d lastB=%d lastTotalOps=%ld lastDone=%d\n",
           r2->tid, r2->ops, r2->snapX, r2->snapY, r2->snapTotal, r2->snapDone);
    printf("Thread %d ops=%ld lastA=%d lastB=%d lastTotalOps=%ld lastDone=%d\n",
           r3->tid, r3->ops, r3->snapX, r3->snapY, r3->snapTotal, r3->snapDone);

    printf("\nFinal Shared: A=%d B=%d total_ops=%ld done=%d\n",
           resX, resY, total_count, finished);
    printf("Expected: total_ops=%ld done=1\n", expected);

    free(r1);
    free(r2);
    free(r3);

    return (total_count == expected && finished == 1) ? 0 : 1;
}
```

---

## Key Design Decisions

| Decision | Reason |
|---|---|
| Always acquire `mxA` before `mxB` | Consistent lock ordering prevents deadlock between `workerXY` and `workerYX` |
| Both workers acquire both locks | `resX`, `resY`, and `total_count` are all updated atomically together |
| Monitor acquires both locks for snapshot | Ensures a consistent read of all shared variables |
| `finished = 1` set under both locks | Prevents a torn read by the monitor thread |
| `build_result()` called under both locks | Snapshot taken atomically after loop ends |
