Looking at this problem, I need to add mutex locks/unlocks to `notxv6/pthread_locks.c` to fix race conditions without causing deadlocks.

Key constraints:
- Only use `lockA` (for resourceA) and `lockB` (for resourceB)
- No deadlock: always lock in same order (lockA before lockB)
- Protect `total_ops` and `done` as well - I'll use lockA for total_ops/done since they need a consistent lock

Let me analyze the locking strategy:
- `worker_AB`: locks A then B (natural order)
- `worker_BA`: also must lock A then B (to avoid deadlock, even though it logically touches B first)
- `worker_monitor`: locks A then B for consistent snapshot

```c
// pthread_locks.c
//
// Compile: gcc -O2 -Wall -Wextra -pthread pthread_locks.c -o student
// Run: ./pthread_locks
//
// Fixed version: Added pthread_mutex_lock()/pthread_mutex_unlock() calls
// to eliminate race conditions and prevent deadlocks.
// Locking discipline: always acquire lockA before lockB to prevent deadlock.

#define _GNU_SOURCE
#include <pthread.h>
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>

// -------------------- Shared Global Resources --------------------

static int sharedValA = 0;   // renamed from resourceA
static int sharedValB = 0;   // renamed from resourceB

static long totalOperations = 0;  // renamed from total_ops
static int finished = 0;          // renamed from done

// Provided mutexes (must use ONLY these)
static pthread_mutex_t lockA = PTHREAD_MUTEX_INITIALIZER;  // protects sharedValA and totalOperations/finished
static pthread_mutex_t lockB = PTHREAD_MUTEX_INITIALIZER;  // protects sharedValB

#define LOOP_COUNT 20000

// -------------------- Thread Argument / Result Types --------------------

typedef struct {
    int thread_id;
} thread_args_t;

typedef struct {
    int thread_id;
    long ops_done;
    int lastA;
    int lastB;
    long lastTotalOps;
    int lastDone;
} thread_result_t;

static void delay(void)
{
    // Fixed micro delay to encourage scheduling interleavings
    usleep(1);
}

static thread_result_t* make_result(int id, long ops)
{
    thread_result_t* resultPtr = malloc(sizeof(thread_result_t));
    if (!resultPtr) { perror("malloc"); exit(1); }

    resultPtr->thread_id = id;
    resultPtr->ops_done = ops;

    // Take a consistent snapshot under both locks (lockA then lockB)
    pthread_mutex_lock(&lockA);
    pthread_mutex_lock(&lockB);

    resultPtr->lastA = sharedValA;
    resultPtr->lastB = sharedValB;
    resultPtr->lastTotalOps = totalOperations;
    resultPtr->lastDone = finished;

    pthread_mutex_unlock(&lockB);
    pthread_mutex_unlock(&lockA);

    return resultPtr;
}

// -------------------- Worker 1: touches A then B --------------------

void* worker_AB(void* arg)
{
    thread_args_t* workerArgs = (thread_args_t*)arg;
    long opsCount = 0;

    for (int idx = 0; idx < LOOP_COUNT && !finished; idx++) {
        // Lock A first, then B (consistent lock ordering to prevent deadlock)
        pthread_mutex_lock(&lockA);
        pthread_mutex_lock(&lockB);

        sharedValA++;          // increment shared resource A
        sharedValB++;          // increment shared resource B
        totalOperations += 2;  // track total operations
        opsCount += 2;

        pthread_mutex_unlock(&lockB);
        pthread_mutex_unlock(&lockA);

        delay();
    }

    return make_result(workerArgs->thread_id, opsCount);
}

// -------------------- Worker 2: touches B then A (logic order) --------------------

void* worker_BA(void* arg)
{
    thread_args_t* workerArgs = (thread_args_t*)arg;
    long opsCount = 0;

    for (int idx = 0; idx < LOOP_COUNT && !finished; idx++) {
        // Must still lock A before B to maintain consistent order and avoid deadlock
        pthread_mutex_lock(&lockA);
        pthread_mutex_lock(&lockB);

        sharedValB += 2;       // increment shared resource B by 2
        sharedValA += 2;       // increment shared resource A by 2
        totalOperations += 2;  // track total operations
        opsCount += 2;

        pthread_mutex_unlock(&lockB);
        pthread_mutex_unlock(&lockA);

        delay();
    }

    return make_result(workerArgs->thread_id, opsCount);
}

// -------------------- Worker 3: monitor (READ-ONLY snapshot) --------------------

void* worker_monitor(void* arg)
{
    thread_args_t* workerArgs = (thread_args_t*)arg;
    long opsCount = 0;

    while (!finished) {
        // Acquire both locks for a consistent snapshot of all shared variables
        pthread_mutex_lock(&lockA);
        pthread_mutex_lock(&lockB);

        int snapA    = sharedValA;
        int snapB    = sharedValB;
        long snapOps = totalOperations;
        int snapDone = finished;

        pthread_mutex_unlock(&lockB);
        pthread_mutex_unlock(&lockA);

        // Print periodically based on total ops snapshot
        if ((snapOps % 20000) == 0) {
            printf("[monitor] A=%d B=%d total_ops=%ld done=%d\n",
                   snapA, snapB, snapOps, snapDone);
        }

        delay();
    }

    return make_result(workerArgs->thread_id, opsCount);
}

// -------------------- Main --------------------

int main(void)
{
    pthread_t threadOne, threadTwo, threadThree;

    thread_args_t argOne = { 1 };
    thread_args_t argTwo = { 2 };
    thread_args_t argThree = { 3 };

    // Create all three threads
    pthread_create(&threadOne,   NULL, worker_AB,      &argOne);
    pthread_create(&threadTwo,   NULL, worker_BA,      &argTwo);
    pthread_create(&threadThree, NULL, worker_monitor, &argThree);

    thread_result_t *resOne = NULL, *resTwo = NULL, *resThree = NULL;

    // Join worker threads first, then signal done
    pthread_join(threadOne, (void**)&resOne);
    pthread_join(threadTwo, (void**)&resTwo);

    // Signal monitor to stop after workers are done
    pthread_mutex_lock(&lockA);
    finished = 1;
    pthread_mutex_unlock(&lockA);

    // Now join the monitor thread
    pthread_join(threadThree, (void**)&resThree);

    const long expectedTotal = 4L * LOOP_COUNT;

    printf("\n--- Results (returned to main) ---\n");
    printf("Thread %d ops=%ld lastA=%d lastB=%d lastTotalOps=%ld lastDone=%d\n",
           resOne->thread_id, resOne->ops_done,
           resOne->lastA, resOne->lastB,
           resOne->lastTotalOps, resOne->lastDone);
    printf("Thread %d ops=%ld lastA=%d lastB=%d lastTotalOps=%ld lastDone=%d\n",
           resTwo->thread_id, resTwo->ops_done,
           resTwo->lastA, resTwo->lastB,
           resTwo->lastTotalOps, resTwo->lastDone);
    printf("Thread %d ops=%ld lastA=%d lastB=%d lastTotalOps=%ld lastDone=%d\n",
           resThree->thread_id, resThree->ops_done,
           resThree->lastA, resThree->lastB,
           resThree->lastTotalOps, resThree->lastDone);

    printf("\nFinal Shared: A=%d B=%d total_ops=%ld done=%d\n",
           sharedValA, sharedValB, totalOperations, finished);
    printf("Expected: total_ops=%ld done=1\n", expectedTotal);

    free(resOne);
    free(resTwo);
    free(resThree);

    // Exit 0 only if deterministic expectations are met
    return (totalOperations == expectedTotal && finished == 1) ? 0 : 1;
}
```
