def assert_equal(got, expect, msg=""):
    if got == expect:
        return
    if msg:
        msg += "\n"
    raise AssertionError("%sgot:\n  %s\nexpected:\n  %s" %
                         (msg, str(got).replace("\n", "\n  "),
                          str(expect).replace("\n", "\n  ")))

def assert_lines_match(txt, *rxps, **kw):
    def _kw(no=[]):
        return no
    no = _kw(**kw)

    lns = txt.splitlines()
    good = set()
    bad = set()
    for i, ln in enumerate(lns):
        if any(re.match(r, ln) for r in rxps):
            good.add(i)
            rxps = [r for r in rxps if not re.match(r, ln)]
        if any(re.match(r, ln) for r in no):
            bad.add(i)

    if not rxps and not bad:
        return

    show = set()
    for lineno in good.union(bad):
        for off in range(-2, 3):
            show.add(lineno + off)
    if rxps:
        show.update(n for n in range(len(lns) - 5, len(lns)))

    msg = []
    last = -1
    for lineno in sorted(show):
        if 0 <= lineno < len(lns):
            if lineno != last + 1:
                msg.append("...")
            last = lineno
            msg.append("%s %s" % (clr("red", "BAD ") if lineno in bad else
                                  clr("green", "GOOD") if lineno in good
                                  else "    ",
                                  lns[lineno]))
    if last != len(lns) - 1:
        msg.append("...")
    if bad:
        msg.append("unexpected lines in output")
    for r in rxps:
        msg.append(clr("red", "MISSING") + " '%s'" % r)
    raise AssertionError("\n".join(msg))

class QEMU(object):
    _GDBPORT = None

    def __init__(self, *mk_args):
        try:
            GDBClient(self.get_gdb_port(), timeout=0).close()
        except socket.error:
            pass
        else:
            print("""\
GDB stub found on port %d.
QEMU appears to already be running.  Please exit it if possible or use
'killall qemu' or 'killall qemu.real'.""" % self.get_gdb_port(), file=sys.stderr)
            sys.exit(1)

        if gopts.verbose:
            show_cmd(("make",) + mk_args)
        cmd = ("make", "-s", "--no-print-directory") + mk_args
        self.proc = Popen(cmd, stdout=subprocess.PIPE,
                          stderr=subprocess.STDOUT,
                          stdin=subprocess.PIPE)
        self.output = ""
        self.outbytes = bytearray()
        self.on_output = []

    @staticmethod
    def get_gdb_port():
        if QEMU._GDBPORT is None:
            p = Popen(["make", "-s", "--no-print-directory", "print-gdbport"],
                      stdout=subprocess.PIPE)
            (out, _) = p.communicate()
            if p.returncode:
                raise RuntimeError(
                    "Failed to get gdbport: make exited with %d" %
                    p.returncode)
            QEMU._GDBPORT = int(out)
        return QEMU._GDBPORT

    def fileno(self):
        if self.proc:
            return self.proc.stdout.fileno()

    def handle_read(self):
        buf = os.read(self.proc.stdout.fileno(), 4096)
        self.outbytes.extend(buf)
        self.output = self.outbytes.decode("utf-8", "replace")
        for cb in self.on_output:
            cb(buf)
        if buf == b"":
            self.wait()
            return

    def write(self, buf):
        if isinstance(buf, str):
            buf = buf.encode('utf-8')
        self.proc.stdin.write(buf)
        self.proc.stdin.flush()

    def wait(self):
        if self.proc:
            self.proc.wait()
            self.proc = None

    def kill(self):
        if self.proc:
            self.proc.terminate()

class GDBClient(object):
    def __init__(self, port, timeout=15):
        t0 = time.time()
        while True:
            self.sock = socket.socket()
            try:
                self.sock.settimeout(1)
                self.sock.connect(("localhost", port))
                break
            except socket.error:
                if time.time() >= t0 + timeout:
                    raise
        self.__buf = ""

    def fileno(self):
        if self.sock:
            return self.sock.fileno()

    def handle_read(self):
        try:
            data = self.sock.recv(4096).decode("ascii", "replace")
        except socket.error:
            data = ""
        if data == "":
            self.sock.close()
            self.sock = None
            return
        self.__buf += data

        while True:
            m = re.search(r"\$([^#]*)#[0-9a-zA-Z]{2}", self.__buf)
            if not m:
                break
            pkt = m.group(1)
            self.__buf = self.__buf[m.end():]

            if pkt.startswith("T05"):
                raise TerminateTest

    def __send(self, cmd):
        pkt = "$%s#%02x" % (cmd, sum(map(ord, cmd)) % 256)
        self.sock.sendall(pkt.encode("ascii"))

    def __send_break(self):
        self.sock.sendall(b"\x03")

    def close(self):
        if self.sock:
            self.sock.close()
            self.sock = None

    def cont(self):
        self.__send("c")

    def breakpoint(self, addr):
        self.__send("Z1,%x,1" % addr)

class TerminateTest(Exception):
    pass

class Runner():
    def __init__(self, *dflt_monitors):
        self.__dflt_monitors = dflt_monitors

    def run_qemu(self, *monitors, **kw):
        def _kw(target_base="qemu", make_args=[], timeout=30):
            return target_base, make_args, timeout
        tgt_base, mk_args, tout = _kw(**kw)

        pre_make()
        self.qemu = QEMU(tgt_base + "-gdb", *mk_args)
        self.gdb = None

        try:
            self.qemu.on_output = [self.__mon_start]
            self.__react([self.qemu], timeout=90)
            self.qemu.on_output = []
            if self.gdb is None:
                print("Failed to connect to QEMU; output:")
                print(self.qemu.output)
                sys.exit(1)
            post_make()

            self.reactors = [self.qemu, self.gdb]

            for m in self.__dflt_monitors + monitors:
                m(self)

            self.gdb.cont()
            self.__react(self.reactors, tout)
        finally:
            try:
                if self.gdb is None:
                    sys.exit(1)
                self.qemu.kill()
                self.__react(self.reactors, 5)
                self.gdb.close()
                self.qemu.wait()
            except:
                print("""\
Failed to shutdown QEMU.  You might need to 'killall qemu' or
'killall qemu.real'.
""")
                raise

    def __mon_start(self, output):
        if b"\n" in output:
            try:
                self.gdb = GDBClient(self.qemu.get_gdb_port(), timeout=2)
                raise TerminateTest
            except socket.error:
                pass
        if not len(output):
            raise TerminateTest

    def __react(self, reactors, timeout):
        deadline = time.time() + timeout
        try:
            while True:
                tleft = deadline - time.time()
                if tleft < 0:
                    sys.stdout.write("Timeout! ")
                    sys.stdout.flush()
                    return

                rset = [r for r in reactors if r.fileno() is not None]
                if not rset:
                    return

                rset, _, _ = select.select(rset, [], [], tleft)
                for r in rset:
                    r.handle_read()
        except TerminateTest:
            pass

    def user_test(self, binary, *monitors, **kw):
        maybe_unlink("obj/kern/init.o", "obj/kern/kernel")
        if kw.pop("snapshot", True):
            kw.setdefault("make_args", []).append("QEMUEXTRA+=-snapshot")
        self.run_qemu(target_base="run-%s" % binary, *monitors, **kw)

    def match(self, *args, **kwargs):
        assert_lines_match(self.qemu.output, *args, **kwargs)

def save(fpath):
    def setup_save(runner):
        f.seek(0)
        f.truncate()
        runner.qemu.on_output.append(f.write)
        get_current_test().on_finish.append(on_fin)

    def on_fin(fail):
        f.flush()
        sp = fpath + "." + get_current_test().__name__[5:]
        if fail:
            shutil.copyfile(fpath, sp)
            print("    QEMU output saved to %s" % sp)
        elif os.path.exists(sp):
            os.unlink(sp)
            print("    (Old %s failure log removed)" % sp)

    f = open(fpath, "wb")
    return setup_save

def stop_breakpoint(addr):
    def setup_bp(runner):
        if isinstance(addr, str):
            addrs = [int(sym[:16], 16) for sym in open("kernel/kernel.sym")
                     if sym[17:].strip() == addr]
            assert len(addrs), "Symbol %s not found" % addr
            runner.gdb.breakpoint(addrs[0])

            runner.gdb.breakpoint(addr)
    return setup_bp

def call_on_line(rxp, cb):
    def setup_col(runner):
        buf = bytearray()
        def handle_out(output):
            buf.extend(output)
            while b"\n" in buf:
                line, buf[:] = buf.split(b"\n", 1)
                line = line.decode("utf-8", "replace")
                if re.match(rxp, line):
                    cb(line)
        runner.qemu.on_output.append(handle_out)
    return setup_col

def stop_on_line(rxp):
    def _stop(line):
        raise TerminateTest
    return call_on_line(rxp, _stop)

def shell_script(script, terminate_match=None):
    def setup_ss(runner):
        class ctx:
            n = 0
            buf = bytearray()
        def handle_out(output):
            ctx.buf.extend(output)
            if terminate_match is not None:
                if re.match(terminate_match, ctx.buf.decode('utf-8', 'replace')):
                    raise TerminateTest
            if b'$ ' in ctx.buf:
                ctx.buf = bytearray()
                if ctx.n < len(script):
                    runner.qemu.write(script[ctx.n])
                    runner.qemu.write('\n')
                    ctx.n += 1
                else:
                    if terminate_match is None:
                        raise TerminateTest
        runner.qemu.on_output.append(handle_out)
    return setup_ss

struct {
  struct spinlock lk;
  struct buf bufs[NBUF];
  struct buf hd;
} blkcache;

void
blkcache_init(void)
{
  struct buf *b;

  initlock(&blkcache.lk, "bcache");

  blkcache.hd.prev = &blkcache.hd;
  blkcache.hd.next = &blkcache.hd;
  for(b = blkcache.bufs; b < blkcache.bufs+NBUF; b++){
    b->next = blkcache.hd.next;
    b->prev = &blkcache.hd;
    initsleeplock(&b->lock, "buffer");
    blkcache.hd.next->prev = b;
    blkcache.hd.next = b;
  }
}

static struct buf*
blk_get(uint dev, uint blockno)
{
  struct buf *b;

  acquire(&blkcache.lk);

  for(b = blkcache.hd.next; b != &blkcache.hd; b = b->next){
    if(b->dev == dev && b->blockno == blockno){
      b->refcnt++;
      release(&blkcache.lk);
      acquiresleep(&b->lock);
      return b;
    }
  }

  for(b = blkcache.hd.prev; b != &blkcache.hd; b = b->prev){
    if(b->refcnt == 0) {
      b->dev = dev;
      b->blockno = blockno;
      b->valid = 0;
      b->refcnt = 1;
      release(&blkcache.lk);
      acquiresleep(&b->lock);
      return b;
    }
  }
  panic("bget: no buffers");
}

struct buf*
blk_read(uint dev, uint blockno)
{
  struct buf *b;

  b = blk_get(dev, blockno);
  if(!b->valid) {
    virtio_disk_rw(b, 0);
    b->valid = 1;
  }
  return b;
}

void
blk_write(struct buf *b)
{
  if(!holdingsleep(&b->lock))
    panic("bwrite");
  virtio_disk_rw(b, 1);
}

void
blk_release(struct buf *b)
{
  if(!holdingsleep(&b->lock))
    panic("brelse");

  releasesleep(&b->lock);

  acquire(&blkcache.lk);
  b->refcnt--;
  if (b->refcnt == 0) {
    b->next->prev = b->prev;
    b->prev->next = b->next;
    b->next = blkcache.hd.next;
    b->prev = &blkcache.hd;
    blkcache.hd.next->prev = b;
    blkcache.hd.next = b;
  }
  
  release(&blkcache.lk);
}

void
blk_pin(struct buf *b) {
  acquire(&blkcache.lk);
  b->refcnt++;
  release(&blkcache.lk);
}

void
blk_unpin(struct buf *b) {
  acquire(&blkcache.lk);
  b->refcnt--;
  release(&blkcache.lk);
}

#define BKSP 0x100
#define CTRL(x)  ((x)-'@')

void
con_putc(int c)
{
  if(c == BKSP){
    uartputc_sync('\b'); uartputc_sync(' '); uartputc_sync('\b');
  } else {
    uartputc_sync(c);
  }
}

struct {
  struct spinlock lk;
  
#define INBUF_SZ 128
  char buf[INBUF_SZ];
  uint r;
  uint w;
  uint e;
} con;

int
con_write(int user_src, uint64 src, int n)
{
  char tmp[32];
  int i = 0;

  while(i < n){
    int nn = sizeof(tmp);
    if(nn > n - i)
      nn = n - i;
    if(either_copyin(tmp, user_src, src+i, nn) == -1)
      break;
    uartwrite(tmp, nn);
    i += nn;
  }

  return i;
}

int
con_read(int user_dst, uint64 dst, int n)
{
  uint tgt;
  int c;
  char cbuf;

  tgt = n;
  acquire(&con.lk);
  while(n > 0){
    while(con.r == con.w){
      if(killed(myproc())){
        release(&con.lk);
        return -1;
      }
      sleep(&con.r, &con.lk);
    }

    c = con.buf[con.r++ % INBUF_SZ];

    if(c == CTRL('D')){
      if(n < tgt){
        con.r--;
      }
      break;
    }

    cbuf = c;
    if(either_copyout(user_dst, dst, &cbuf, 1) == -1)
      break;

    dst++;
    --n;

    if(c == '\n'){
      break;
    }
  }
  release(&con.lk);

  return tgt - n;
}

void
con_intr(int c)
{
  acquire(&con.lk);

  switch(c){
  case CTRL('P'):
    procdump();
    break;
  case CTRL('U'):
    while(con.e != con.w &&
          con.buf[(con.e-1) % INBUF_SZ] != '\n'){
      con.e--;
      con_putc(BKSP);
    }
    break;
  case CTRL('H'):
  case '\x7f':
    if(con.e != con.w){
      con.e--;
      con_putc(BKSP);
    }
    break;
  default:
    if(c != 0 && con.e-con.r < INBUF_SZ){
      c = (c == '\r') ? '\n' : c;

      con_putc(c);

      con.buf[con.e++ % INBUF_SZ] = c;

      if(c == '\n' || c == CTRL('D') || con.e-con.r == INBUF_SZ){
        con.w = con.e;
        wakeup(&con.r);
      }
    }
    break;
  }
  
  release(&con.lk);
}

void
con_init(void)
{
  initlock(&con.lk, "cons");

  uartinit();

  devsw[CONSOLE].read = con_read;
  devsw[CONSOLE].write = con_write;
}
