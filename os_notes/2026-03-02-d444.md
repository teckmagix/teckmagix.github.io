**notxv6/pthread_locks.c**

```c
#define _GNU_SOURCE
#include <pthread.h>
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>

static int valX = 0;
static int valY = 0;

static long total_count = 0;
static int finished = 0;

static pthread_mutex_t mtxA = PTHREAD_MUTEX_INITIALIZER;
static pthread_mutex_t mtxB = PTHREAD_MUTEX_INITIALIZER;

#define ITER_MAX 20000

typedef struct {
    int tid;
} targs_t;

typedef struct {
    int tid;
    long cnt;
    int snapX;
    int snapY;
    long snapTotal;
    int snapDone;
} tres_t;

static void pause_brief(void)
{
    usleep(1);
}

static tres_t* build_result(int id, long ops)
{
    tres_t* r = malloc(sizeof(tres_t));
    if (!r) { perror("malloc"); exit(1); }

    r->tid = id;
    r->cnt = ops;

    pthread_mutex_lock(&mtxA);
    pthread_mutex_lock(&mtxB);
    r->snapX = valX;
    r->snapY = valY;
    r->snapTotal = total_count;
    r->snapDone = finished;
    pthread_mutex_unlock(&mtxB);
    pthread_mutex_unlock(&mtxA);

    return r;
}

void* task_XY(void* arg)
{
    targs_t* p = (targs_t*)arg;
    long ops = 0;

    for (int i = 0; i < ITER_MAX && !finished; i++) {
        pthread_mutex_lock(&mtxA);
        pthread_mutex_lock(&mtxB);

        valX++;
        valY++;
        total_count += 2;
        ops += 2;

        pthread_mutex_unlock(&mtxB);
        pthread_mutex_unlock(&mtxA);

        pause_brief();
    }

    return build_result(p->tid, ops);
}

void* task_YX(void* arg)
{
    targs_t* p = (targs_t*)arg;
    long ops = 0;

    for (int i = 0; i < ITER_MAX && !finished; i++) {
        pthread_mutex_lock(&mtxA);
        pthread_mutex_lock(&mtxB);

        valY += 2;
        valX += 2;
        total_count += 2;
        ops += 2;

        pthread_mutex_unlock(&mtxB);
        pthread_mutex_unlock(&mtxA);

        pause_brief();
    }

    return build_result(p->tid, ops);
}

void* task_watch(void* arg)
{
    targs_t* p = (targs_t*)arg;
    long ops = 0;

    while (!finished) {
        pthread_mutex_lock(&mtxA);
        pthread_mutex_lock(&mtxB);

        int a = valX;
        int b = valY;
        long t = total_count;
        int d = finished;

        pthread_mutex_unlock(&mtxB);
        pthread_mutex_unlock(&mtxA);

        if ((t % 20000) == 0) {
            printf("[monitor] A=%d B=%d total_ops=%ld done=%d\n", a, b, t, d);
        }

        pause_brief();
    }

    return build_result(p->tid, ops);
}

int main(void)
{
    pthread_t w1, w2, w3;

    targs_t g1 = { 1 };
    targs_t g2 = { 2 };
    targs_t g3 = { 3 };

    pthread_create(&w1, NULL, task_XY, &g1);
    pthread_create(&w2, NULL, task_YX, &g2);
    pthread_create(&w3, NULL, task_watch, &g3);

    tres_t *r1 = NULL, *r2 = NULL, *r3 = NULL;

    pthread_join(w1, (void**)&r1);
    pthread_join(w2, (void**)&r2);

    pthread_mutex_lock(&mtxA);
    finished = 1;
    pthread_mutex_unlock(&mtxA);

    pthread_join(w3, (void**)&r3);

    const long expected = 4L * ITER_MAX;

    printf("\n--- Results (returned to main) ---\n");
    printf("Thread %d ops=%ld lastA=%d lastB=%d lastTotalOps=%ld lastDone=%d\n",
           r1->tid, r1->cnt, r1->snapX, r1->snapY, r1->snapTotal, r1->snapDone);
    printf("Thread %d ops=%ld lastA=%d lastB=%d lastTotalOps=%ld lastDone=%d\n",
           r2->tid, r2->cnt, r2->snapX, r2->snapY, r2->snapTotal, r2->snapDone);
    printf("Thread %d ops=%ld lastA=%d lastB=%d lastTotalOps=%ld lastDone=%d\n",
           r3->tid, r3->cnt, r3->snapX, r3->snapY, r3->snapTotal, r3->snapDone);

    printf("\nFinal Shared: A=%d B=%d total_ops=%ld done=%d\n",
           valX, valY, total_count, finished);
    printf("Expected: total_ops=%ld done=1\n", expected);

    free(r1);
    free(r2);
    free(r3);

    return (total_count == expected && finished == 1) ? 0 : 1;
}
```
