## File 2: quiz2-q3.zip — `user/uthread.c` (Complete Solution)

```c
#include "kernel/types.h"
#include "kernel/stat.h"
#include "user/user.h"
#include "user/uthread.h"

struct thread all_thread[MAX_THREAD];
struct thread *current_thread;

void 
thread_init(void)
{
  // main() is thread 0, which will make the first invocation to
  // thread_schedule(). It needs a stack so that the first thread_switch() can
  // save thread 0's state.
  current_thread = &all_thread[0];
  current_thread->state = RUNNING;
}

void 
thread_sleep()
{
  current_thread->state = SLEEPING;
  thread_schedule();
}

int 
thread_wakeup(int thread_id)
{
  // Validate thread_id: must be in range [0, MAX_THREAD)
  if (thread_id < 0 || thread_id >= MAX_THREAD) {
    return -1;
  }
  // Only wake up if the thread is SLEEPING
  if (all_thread[thread_id].state == SLEEPING) {
    all_thread[thread_id].state = RUNNABLE;
    return 0;
  }
  return -1;
}

void thread_schedule(void)
{
  struct thread *t, *next_thread;
  int sleeping_thread = 0;

  /* Find another runnable thread. */
  next_thread = 0;
  t = current_thread + 1;
  for(int i = 0; i < MAX_THREAD; i++){
    if(t >= all_thread + MAX_THREAD)
      t = all_thread;
    if(t->state == RUNNABLE) {
      next_thread = t;
      break;
    }
    // identify that there are sleeping threads
    if(t->state == SLEEPING) {
      sleeping_thread = 1;
    }
    t = t + 1;
  }

  if (next_thread == 0) {
    // No runnable thread. If there is a sleeping thread, wake it up. Otherwise, panic.
    if (sleeping_thread == 1){
      printf("only thread sleepings, wake them all\n");
      // we use first_sleep to make sure only the first sleeping thread is set to RUNNING,
      // and the rest are set to RUNNABLE, so that they can be scheduled in the future.
      int first_sleep = 0;
      for(int i = 0; i < MAX_THREAD; i++){
        // check if thread i is sleeping
        if(all_thread[i].state == SLEEPING){
          if (first_sleep == 0){
            // set the first sleeping thread as next to be executed
            next_thread = &all_thread[i];
            all_thread[i].state = RUNNABLE;
            first_sleep = 1;
          } else {
            // wake up remaining sleeping threads (set to RUNNABLE)
            all_thread[i].state = RUNNABLE;
          }
        }
      }
    }
    else{
      printf("thread_schedule: no runnable threads\n");
      exit(-1);
    }
  }

  if (current_thread != next_thread) {         /* switch threads?  */
    next_thread->state = RUNNING;
    t = current_thread;
    current_thread = next_thread;
    thread_switch(&t->context, &current_thread->context);
  } else
    next_thread = 0;
}

void 
thread_create(void (*func)())
{
  struct thread *t;

  for (t = all_thread; t < all_thread + MAX_THREAD; t++) {
    if (t->state == FREE) break;
  }
  t->state = RUNNABLE;
  t->context.sp = (uint64)&t->stack[STACK_SIZE-1];
  t->context.ra = (uint64)(*func);
}

void 
thread_yield(void)
{
  current_thread->state = RUNNABLE;
  thread_schedule();
}
```

> **💡 Hint:** Threads are indexed by their position in `all_thread[]`, so `thread_id` directly maps to `all_thread[thread_id]`. The scheduler scans circularly starting from `current_thread + 1`.

> **⚠️ Warning:** The `sleeping_thread` flag must be set inside the loop **before** breaking out — make sure the `SLEEPING` check happens even when no `RUNNABLE` thread is found yet. Also, `thread_wakeup` must validate both bounds (`< 0` and `>= MAX_THREAD`) to handle the `thread_wakeup(-1)` and `thread_wakeup(4)` error cases tested in `uthread_ex1.c`.

**Explanation of each function:**

- **`thread_sleep()`**: Sets `current_thread->state = SLEEPING` then calls `thread_schedule()` to hand off the CPU to the next runnable thread.

- **`thread_wakeup(int thread_id)`**: Validates the index is within `[0, MAX_THREAD)`, then checks if `all_thread[thread_id].state == SLEEPING`. If sleeping, sets it to `RUNNABLE` and returns `0`; otherwise returns `-1`.

- **`thread_schedule()` modifications**:
  1. Inside the loop, when a thread is `SLEEPING`, set `sleeping_thread = 1`.
  2. In the `next_thread == 0` branch, iterate all threads; the first `SLEEPING` thread becomes `next_thread` (set `RUNNABLE`), all subsequent sleeping threads are also set to `RUNNABLE`.
